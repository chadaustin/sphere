System Data Types
-----------------
struct image
{
  int width, height;
}

struct color
{
  int red, green, blue, alpha;  
}

struct point
{
  int x, y;
}



System Functions
----------------
- NOTE: Internally, all function names (system labels) begin with an underscore.
-       For example, Exit() is _Exit


0x00    void  Exit()
0x01    void  ExitMessage(string message)
0x02    void  Delay(int milliseconds)
0x03    void  SetGameFont(string font)
0x04    void  SetGameWindowStyle(string windowstyle)
0x05    int   GetTime()

0x50    void  ChangeMap(string map)
0x51    void  MapEngine()
0x52    void  SetColorMask(color mask, int milliseconds)

0x60    void  AddPartyCharacter(string spriteset)

0x10    void  FlipScreen()
0x11    void  SetClippingRectangle(int x, int y, int w, int h)
0x12    image LoadImage(string file)
0x13    image GrabImage(int x, int y, int w, int h)
0x14    void  DestroyImage(image i)
0x15    void  BlitImage(image i, int x, int y)
0x16    void  ApplyColorMask(int level, color mask)

0x17    void  FadeIn(int milliseconds)
0x18    void  FadeOut(int milliseconds)

0x19    void  DrawText(int x, int y, string text)
0x1A    void  DrawTextBox(int x, int y, int w, int h, int offset, string text)
0x1B    void  DrawWindow(int x, int y, int w, int h)
0x1C    void  SetTextColor(color c)

0x20    int   PlaySound(string file)
0x21    int   ChangeSong(string file)

0x30    void  RefreshInput()
0x31    int   AnyKeyPressed()
0x32    int   KeyPressed(int key)

0x40    string MidString(string s, int offset, int numchars)
0x41    string LeftString(string s, int numchars)
0x42    string RightString(string s, int numchars)
0x43    int    StringLength(string s)

0x70    void PlayAnimation(string file)
